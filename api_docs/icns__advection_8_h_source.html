561  }
562 
563  //
564  // Predict velocities
565  //
566 
567  for (int lev = 0; lev < repo.num_active_levels(); ++lev) {
568  MOL::ExtrapVelToFaces(
569  dof_field(lev), u_mac(lev), v_mac(lev), w_mac(lev),
570  repo.mesh().Geom(lev), dof_field.bcrec(),
571  dof_field.bcrec_device().data());
572  }
573 
574  m_macproj_op(fstate, dt);
575  }

577  void operator()(const FieldState fstate, const amrex::Real /*unused*/)
578  {
579 
580  const auto& repo = fields.repo;
581  const auto& geom = repo.mesh().Geom();
582  // cppcheck-suppress constVariable
583  auto& conv_term = fields.conv_term.state(fstate);
584  const auto& dof_field = fields.field.state(fstate);
585  const auto& rho = repo.get_field("density").state(fstate);
586 
587  //
588  // Advect velocity
589  //
590 
591  int nmaxcomp = AMREX_SPACEDIM;
592  for (int lev = 0; lev < repo.num_active_levels(); ++lev) {
593 
594  amrex::MFItInfo mfi_info;
595  // if (amrex::Gpu::notInLaunchRegion())
596  // mfi_info.EnableTiling(amrex::IntVect(1024,16,16)).SetDynamic(true);
597  if (amrex::Gpu::notInLaunchRegion()) {
598  mfi_info.EnableTiling(amrex::IntVect(1024, 1024, 1024))
599  .SetDynamic(true);
600  }
601 #ifdef AMREX_USE_OMP
602 #pragma omp parallel if (amrex::Gpu::notInLaunchRegion())
603 #endif
604  for (amrex::MFIter mfi(dof_field(lev), mfi_info); mfi.isValid();
605  ++mfi) {
606  amrex::Box const& bx = mfi.tilebox();
607  amrex::Box gbx = grow(bx, fvm::MOL::nghost_state);
608 
609  // Set up momentum array
610  amrex::FArrayBox qfab(gbx, ICNS::ndim);
611  const auto& q = qfab.array();
612  // Calculate momentum
613  auto rho_arr = rho(lev).const_array(mfi);
614  auto vel_arr = dof_field(lev).const_array(mfi);
615  amrex::ParallelFor(
616  gbx, ICNS::ndim,
617  [=] AMREX_GPU_DEVICE(int i, int j, int k, int n) noexcept {
618  q(i, j, k, n) = rho_arr(i, j, k) * vel_arr(i, j, k, n);
619  });
620  // Doing this explicitly, instead of through a Multiply command,
621  // helps avoid floating-point errors with intel compilers and
622  // mimics the implementation in equation_systems/AdvOp_MOL.H
623  amrex::Gpu::streamSynchronize();
624 
625  amrex::Box tmpbox = amrex::surroundingNodes(bx);
626  const int tmpcomp = nmaxcomp * AMREX_SPACEDIM;
627 
628  amrex::FArrayBox tmpfab(tmpbox, tmpcomp);
629  amrex::Elixir eli = tmpfab.elixir();
630 
631  amrex::Array4<amrex::Real> fx = tmpfab.array(0);
632  amrex::Array4<amrex::Real> fy = tmpfab.array(nmaxcomp);
633  amrex::Array4<amrex::Real> fz = tmpfab.array(nmaxcomp * 2);
634 
635  mol::compute_convective_fluxes(
636  lev, bx, AMREX_SPACEDIM, fx, fy, fz, q,
637  u_mac(lev).const_array(mfi), v_mac(lev).const_array(mfi),
638  w_mac(lev).const_array(mfi), dof_field.bcrec().data(),
639  dof_field.bcrec_device().data(), geom);
640 
641  mol::compute_convective_rate(
642  bx, AMREX_SPACEDIM, conv_term(lev).array(mfi), fx, fy, fz,
643  geom[lev].InvCellSizeArray());
644  }
645  }
646  }
