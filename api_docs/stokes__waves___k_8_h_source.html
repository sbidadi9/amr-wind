55  if (StokesOrder == 2) {

63  (8 * Sh * std::pow(1 - S, 3));
64  b31 = -3 * (1 + 3 * S + 3 * std::pow(S, 2) + 2 * std::pow(S, 3)) /

72  45 * std::pow(S, 4) - 13 * std::pow(S, 5)) /
73  (24 * std::pow(1 - S, 5));
74  a44 = (10 * std::pow(S, 3) - 174 * std::pow(S, 4) + 291 * std::pow(S, 5) +
75  278 * std::pow(S, 6)) /
76  (48 * (3 + 2 * S) * std::pow(1 - S, 5));
77  b42 = CTh *
78  (6 - 26 * S - 182 * std::pow(S, 2) - 204 * std::pow(S, 3) -
79  25 * std::pow(S, 4) + 26 * std::pow(S, 5)) /
80  (6 * (3 + 2 * S) * std::pow(1 - S, 4));

83  67 * std::pow(S, 4) + 34 * std::pow(S, 5)) /

106  1376 * std::pow(S, 4) - 1302 * std::pow(S, 5) -
107  117 * std::pow(S, 6) + 58 * std::pow(S, 7)) /

111  852 * std::pow(S, 6) + 2029 * std::pow(S, 7) + 430 * std::pow(S, 8)) /
112  (64 * Sh * (3 + 2 * S) * (4 + S) * std::pow(1 - S, 6));
113  b53 = 9 *

115  6292 * std::pow(S, 4) - 2687 * std::pow(S, 5) +
116  194 * std::pow(S, 6) + 467 * std::pow(S, 7) + 82 * std::pow(S, 8)) /
117  (128 * (3 + 2 * S) * (4 + S) * std::pow(1 - S, 6));

121  1188 * std::pow(S, 4) + 675 * std::pow(S, 5) + 1326 * std::pow(S, 6) +
122  827 * std::pow(S, 7) + 130 * std::pow(S, 8)) /
123  (384 * (3 + 2 * S) * (4 + S) * std::pow(1 - S, 6));
